```python
def calculate_series(terms):
    if terms <= 0:
        raise ValueError("The number of terms must be a positive integer.")

    total_sum = 0.0
    for n in range(terms):
        denominator = 2 * n + 1  # The odd numbers: 1, 3, 5, 7, ...
        term = (-1) ** n * (1 / denominator)  # Alternating series
        total_sum += term

    return total_sum * 4  # Multiply the total by 4 as per the series definition

try:
    result = calculate_series(10000)
    print(f"The approximate value of Pi after calculating 10,000 terms is: {result}")
except ValueError as e:
    print(f"Error: {e}")
```
